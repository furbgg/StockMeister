IngredientService.java 

@Transactional
public Ingredient updateIngredient(Long id, Ingredient data) {
    Ingredient existing = repo.findById(id).orElseThrow();
    
    // Check for price changes (Business Logic)
    if (hasPriceChanged(existing, data)) {
        log.info("Price update detected for {}", existing.getName());
        
        // Trigger cascading update
        recipeService.recalculateAll(id);
    }
    
    existing.updateFrom(data);
    return repo.save(existing);
}

SecurityConfig.java 

@Bean
public SecurityFilterChain securityFilterChain(HttpSecurity http) {
    http
        .csrf(csrf -> csrf.disable())
        .sessionManagement(sess -> sess.sessionCreationPolicy(STATELESS))
        .authorizeHttpRequests(auth -> auth
            // Public Endpoints
            .requestMatchers("/api/auth/**").permitAll()
            
            // Protected Logic
            .requestMatchers(HttpMethod.DELETE, "/api/**")
                .hasRole("ADMIN")
            
            .requestMatchers("/api/orders/**")
                .hasAnyRole("WAITER", "CHEF")
                
            .anyRequest().authenticated())
        .addFilterBefore(jwtAuthFilter, UsernamePasswordFilter.class);
        
    return http.build();
}
